\section{Theory to Design}

\begin{frame}{Theoretical Core and Implementation Consequence}
\small
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\begin{block}{Research claim}
Carrier source distribution (real vs ML) may alter steganalysis detectability under matched embedding conditions.
\end{block}
\begin{block}{Design implication}
The pipeline must control everything except source:
\begin{itemize}
\item same image shape and format (RGB, 512x512, PNG)
\item same payload levels and encryption states
\item same detector protocol and grouped folds
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.42\textwidth}
\begin{alertblock}{Operational rule}
\small
Every artifact is keyed by \texttt{group\_id}, and all variants of the same semantic anchor stay grouped through splitting and evaluation.
\end{alertblock}
\begin{exampleblock}{Why this matters}
\small
Prevents leakage and lets us attribute performance differences to source/method/payload/encryption, not data contamination.
\end{exampleblock}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{RQ Mapping to Pipeline Outputs}
\scriptsize
\renewcommand{\arraystretch}{1.18}
\begin{tabular}{@{}p{1.0cm}p{5.0cm}p{6.9cm}@{}}
\toprule
\textbf{RQ} & \textbf{Question} & \textbf{Pipeline output required} \\
\midrule
RQ1 & Real vs pooled ML detectability & Fold-wise prediction tables by source + AUC contrasts \\
RQ2 & ML-A vs ML-B detectability & Source-specific metrics (\texttt{ml\_a}, \texttt{ml\_b}) \\
RQ3 & Payload interaction & Metrics grouped by \texttt{payload\_level} (low/medium/high) \\
RQ4 & Embedding interaction & Metrics grouped by \texttt{method} (lsb/dct) \\
RQ5 & Encryption interaction & Metrics grouped by \texttt{encryption} (plain/encrypted) \\
\bottomrule
\end{tabular}
\vspace{0.15cm}
\begin{block}{Implementation interpretation}
The code must produce consistent condition keys in every manifest and prediction row: 
\texttt{group\_id, source, method, payload\_level, encryption, fold}.
\end{block}
\end{frame}

\section{Locked Contracts}

\begin{frame}{Locked Dataset Cardinalities}
\small
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{Canonical unit}
\texttt{group\_id} = one caption/prompt anchor
\end{block}
\begin{itemize}
\item Groups: \textbf{500}
\item Covers/group: \textbf{3} (real, ml\_a, ml\_b)
\item Total covers: \textbf{1,500}
\item Methods: \textbf{2} (lsb, dct)
\item Payload levels: \textbf{3}
\item Encryption states: \textbf{2}
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\begin{alertblock}{Derived counts}
\small
Stego variants per cover =
\[
2 \times 3 \times 2 = 12
\]
Total stegos =
\[
1{,}500 \times 12 = 18{,}000
\]
Persisted images total =
\[
1{,}500 + 18{,}000 = 19{,}500
\]
\end{alertblock}
\vspace{-0.1cm}
{\scriptsize Consistency check: every cover row maps to exactly 12 stego rows.}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[fragile]{Critical Function Calls Inside runner.py}
\tiny
\begin{lstlisting}[language=Python]
# Standardization stage
standardize_and_save(input_path, output_path, size=config.image_size)

# Payload generation stage
payload_bytes = _generate_payload_bytes(payload_bits, rng)
if encryption == "encrypted":
    ciphertext = encrypt_payload_aes_256_cbc(payload_bytes, key=key, iv=iv)

# Embedding execution stage
if method == "lsb":
    stego = embed_lsb(cover_image, payload_bytes, payload_level, prng_key)
elif method == "dct":
    stego = embed_dct_qim(cover_image, payload_bytes, payload_level, delta)

# Split stage
folds = generate_grouped_5fold_splits(group_ids, seed=split_seed)
\end{lstlisting}
{\scriptsize Deferred algorithms are invoked only here; runner provides paths, payloads, and execution context.}

\section{Training and Evaluation Logic}

\begin{frame}{Grouped 5-Fold Strategy and Counts}
\small
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{Split rule}
Split unit = \texttt{group\_id} (never individual image row)
\end{block}
\begin{itemize}
\item 500 groups total
\item Per fold: 100 test groups
\item Remaining 400: 350 train + 50 validation
\item Prevents leakage across real/ml\_a/ml\_b variants
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Per-method positives per fold}
\scriptsize
Train: \(350 \times 3 \times 6 = 6{,}300\)\\
Val: \(50 \times 3 \times 6 = 900\)\\
Test: \(100 \times 3 \times 6 = 1{,}800\)
\end{block}
\begin{alertblock}{SRM training scope}
\small
Per-method SRM models:\\
\texttt{lsb} model + \texttt{dct} model for each fold\\
\(2 \times 5 = 10\) training runs.
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Detector Policy and Applicability}
\small
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{Confirmatory detectors}
\begin{itemize}
\item SRM+EC (trained, per-method)
\item RS (LSB domain)
\item Block-DCT shift test (DCT domain)
\end{itemize}
\end{block}
\begin{exampleblock}{Sensitivity-only}
\small
\(\chi^2\) for LSB branch diagnostics.
\end{exampleblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Evaluation table key}
\scriptsize
\texttt{fold, detector, group\_id, source, method, payload\_level, encryption, label, score}
\end{block}
\begin{alertblock}{Aggregation levels}
\small
fold -> detector -> condition -> source -> pooled summaries with uncertainty intervals.
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\section{Deferred Contracts}

\begin{frame}[fragile]{Deferred Function Contracts (Closed-Loop)}
\tiny
\begin{lstlisting}[language=Python]
# Encryption
encrypt_payload_aes_256_cbc(payload: bytes, key: bytes, iv: bytes) -> bytes
decrypt_payload_aes_256_cbc(ciphertext: bytes, key: bytes, iv: bytes) -> bytes

# Embedding
embed_lsb(cover_image: Image.Image, payload_bytes: bytes,
          payload_level: str, prng_key: int) -> Image.Image
embed_dct_qim(cover_image: Image.Image, payload_bytes: bytes,
              payload_level: str, delta: float) -> Image.Image
\end{lstlisting}
\begin{alertblock}{Rule}
{\scriptsize No reads/writes inside these functions. Runner supplies inputs and handles persistence.}
\end{alertblock}
\end{frame}

\begin{frame}[fragile]{Deferred Function Contracts: Detection and SRM}
\tiny
\begin{lstlisting}[language=Python]
# Statistical detectors
rs_analysis_score(image: Image.Image) -> float
chi_square_score(image: Image.Image) -> float
block_dct_shift_score(image: Image.Image) -> float

# SRM model API
train_srm_ec_model(training_input: SRMTrainingInput) -> SRMModelArtifact
score_srm_ec_model(model: SRMModelArtifact,
                   x_samples: Sequence[Sequence[float]]) -> list[float]
\end{lstlisting}
\begin{alertblock}{Rule}
{\scriptsize Input/output only. Model training/inference functions return artifacts/scores and do not perform persistence.}
\end{alertblock}
\end{frame}

\begin{frame}{What Each Deferred Component Must Implement}
\small
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{Encryption module}
\scriptsize
Validate key/IV length, apply AES-256-CBC, return bytes; deterministic for fixed key+IV.
\end{block}
\begin{block}{LSB module}
\scriptsize
Map payload level to embedding intensity (k / pixel fraction), deterministic with \texttt{prng\_key}, return stego image.
\end{block}
\begin{block}{DCT module}
\scriptsize
Block DCT + QIM with \texttt{delta}; use payload-level coefficient fraction; return stego image.
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Statistical detectors}
\scriptsize
Consume one in-memory image, return scalar score calibrated for ROC/AUC ranking.
\end{block}
\begin{block}{SRM module}
\scriptsize
Train: feature vectors + labels -> model artifact\\
Score: model artifact + feature vectors -> score vector
\end{block}
\begin{exampleblock}{Acceptance criteria}
\scriptsize
Shape checks, deterministic outputs under fixed seeds, no side effects, test coverage from contract tests.
\end{exampleblock}
\end{column}
\end{columns}
\end{frame}
